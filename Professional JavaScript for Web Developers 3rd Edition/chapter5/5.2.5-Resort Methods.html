1. Array provide two methods to resort: "reverse()" and "sort()"
2. we can custom the way of sort by deliver method to "sort()" 
Test Code:
//Ex1
var arr = [2,4,1,6];
arr.sort();
console.log(arr);
arr.reverse();
console.log(arr);
//Ex2
var arr2 = [3,6,22,7,9];
arr2.sort(); 
console.log(arr2);
//sort base first num of arr's element, 
//not the effect we wanted, custom it.
arr2.sort(function(v1, v2){
    if(v1 < v2)
        return -1;
    else if (v1 > v2)
        return 1;
    else return 0;
});
console.log(arr2);
//sort(custom_function) resort Array base on
// 3 return values of cumstom_function to sort:
// >0 , =0 and <0;
//to above function, we can just write like below.
arr2.sort(function(v1, v2){
    return v1 - v2;
});
console.log(arr2);
