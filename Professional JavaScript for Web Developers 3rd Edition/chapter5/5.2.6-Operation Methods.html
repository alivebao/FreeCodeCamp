1. slice(begin, end), will not change arr itself, but return a new arr
2. splice(begin, end, ele2Add), great funtion,
can used to add/update/remove elements of array. 
Test Code:
//Ex1
var arr1 = [1,4,6,3];
var arr2 = arr1.slice(1,3);
console.log(arr1);
console.log(arr2);
//Ex2
console.log("add at head");
arr1.splice(0, 0, "head");
console.log(arr1);
console.log("del last ele");
arr1.splice(arr1.length - 1, arr1.length);
console.log(arr1);
console.log("update first ele")
arr1.splice(0, 1 ,"updated");
console.log(arr1);
