1. callee: arguments.callee, return the function which arguments belong.
  2. caller: arguments.callee.caller, return the function's caller.
    Return null when caller is global
Test code:
function outer(){
    console.log(outer.caller);
    inner();
}
function inner(){
    console.log(arguments.callee);
    console.log(inner.caller);
    console.log(inner);
}
outer();
