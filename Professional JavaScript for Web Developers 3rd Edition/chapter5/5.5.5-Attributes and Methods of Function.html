1. length: return the count of arguments function excepted.
  2. apply() and call(): set the "this"
    • apply(): except two para, scope and an array(used to set the param of caller)
    • call(): except multiple para, scope and other(used to set the param of caller)
  3. bind(): create a instance of Function, and it's "this" binded to the param called by caller
Test code:
//Ex1
function sayName(name){
    console.log(name);
}
console.log(sayName.length);
//Ex2
window.color = "red";
var o = {color: "green"};
function sayColor(var1, var2){
    console.log(this.color);
    console.log("var1:" + var1);
    console.log("var2:" + var2);
}
sayColor.apply(this,["ha", "var2"]);
sayColor.call(o, "ha", "var2");
//Ex3
var oSayColor = sayColor.bind(o);
oSayColor();
