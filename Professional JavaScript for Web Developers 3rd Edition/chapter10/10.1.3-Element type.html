Element type
  1. HTML element:id/title/lang/dir/className
  2. getAttribute(attrName)

    • xx.getAttribute("style"): return text description css of element in style="xxx"

      xx.style return a css object

    • Event Handler like xx.getAttribute("onclick"): return text in "onclick=xxx"

      xx.onclick: return a JS function

  3. setAttribute(attrName, value), custom attribute need a prex "data-" like: data-attrName
  4. removeAttribute
  5. attributes
  6. create Element: document.createElement("xx")
test code:
var div = document.getElementById("myDiv");
console.log(div.tagName);
console.log(div.tagName == div.nodeName);

console.log(div.getAttribute("class"));
console.log(div.getAttribute("id"));
console.log(div.getAttribute("data-Attri"));

console.log(div.getAttribute("style"));
console.log(div.style);

console.log(div.getAttribute("onclick"));
console.log(div.onclick);

var endDiv = document.createElement("div");
endDiv.setAttribute("style", "background-color: black; height: 20px;");
document.body.appendChild(endDiv);
