This method encapsulation all message in constructor method.
It decide whether prototype init needed by check if a method exist.
In below test code, we can see that "this.sayName" just defined one time. 
Test code:
function Person(name, age, job){
    this.name = name;
    this.age = age;
    this.job = job;    
    
    if(typeof this.sayName != "function"){
        Person.prototype.sayName = function(){
            console.log(this.name);
        }
    }
}
