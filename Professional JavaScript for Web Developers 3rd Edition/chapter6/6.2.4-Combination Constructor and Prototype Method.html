Use constructor method to define instance attribute
Use prototype method to define method and shared attribute
This is a default method to define a Reference Type. 
Test code:
function Person(name, age, job){
    this.name = name;
    this.age = age;
    this.job = job;    
}
Person.prototype = {
    constructor: Person,
    sayName: function(){
        console.log(this.name);
    }
}
